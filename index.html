That's a long name

really-simple-bootstrap-color-picker is a fork of really-simple-color-picker, thus the long name.

Example usage

<form class="form-horizontal">
  <div class="control-group">
    <label class="control-label">Colored label</label>
    <div class="controls input-prepend">
      <input type="text" id="color" name="color" class="input-small" />
    </div>
  </div>
</form>

$(function() {
  var $elem = $('#color');
  $elem.colorPicker({pickerDefault: '#CCFFFF'});

  $target = $elem.parents('.control-group:first').find('label:first');

  $target.css('color', $elem.val());

  $elem.on('colorPicker:preview colorPicker:change', function(e, value) {
    $target.css('color', value);
  });
});
